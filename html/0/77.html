3. css代码

用于设定浮动的位置 
.gotop_btn1 {
position: fixed;
left: 50%;
margin-left: 458px;
top: 400px;
font-weight:700;
font-size:11px;
width: 22px;
height: 71px;
display: none;
}

4. js代码

用于设定在什么情况下浮动标签出现 
window.onscroll = function()
{
	var h =document.body.scrollTop;
	if(!h)
		h=document.documentElement.scrollTop;
	var top = document.getElementById('goTopButton');
	if(h>0)
	{
		top.style.display = 'block';
	}
	else
	{
		top.style.display = 'none';
	}
};
